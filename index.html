Here is my Advent of Code 2022 (day 1) solution:

const fs = require('fs');
const readline = require('readline');
const { Stream } = require('stream');
const input_path = 'input.txt';
const input = fs.createReadStream(input_path);
const output = new Stream;

let lineReader = readline.createInterface(input, output);
let topCalories = [];
let currentCalories = 0;

lineReader.on("line", function(line) {
    if (line.length > 0) {
        currentCalories += Number(line);
    } else {
        topCalories.push(currentCalories);
        currentCalories = 0;
    }
});

lineReader.on('close', function() {
    console.log(topCalories.sort( (a, b) => b - a)[0]);
});
